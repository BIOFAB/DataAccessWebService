/**
 * @class Ext.layout.component.Auto
 * @extends Ext.layout.Component
 *
 * 
The AutoLayout is the default layout manager delegated by {@link Ext.Component} to
 * render any child Elements when no {@link Ext.Component#layout layout} is configured.

 */

Ext.define('Ext.layout.component.Auto', {

    /* Begin Definitions */

    alias: 'layout.autocomponent',

    extend: 'Ext.layout.Component',

    /* End Definitions */

    type: 'autocomponent',

    onLayout : function(width, height) {
        this.setTargetSize(width, height);
    }
});
